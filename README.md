```json
{
  "emmet.triggerExpansionOnTab": true,
  "emmet.showSuggestionsAsSnippets": true,
  "emmet.showExpandedAbbreviation": "always",
  "emmet.variables": {
    "charset": "UTF-8",
    "lang": "ko-KR"
  },
  "emmet.syntaxProfiles": {
    "attr_case": "lower",
    "attr_quotes": "double",
    "html": "xhtml",
    "inline_break": 3,
    "self_closing_tag": "xhtml",
    "tag_case": "lower"
  },
}
```
